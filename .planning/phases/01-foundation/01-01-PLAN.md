    ```typescript
    "use client";

    import React, { useState, useEffect, useCallback } from "react";
    import Split from "react-split";
    import { Card, Typography, Space, Flex } from "antd";
    import { useTranslations } from "next-intl";
    import { useLocalStorage } from "@/app/hooks/useLocalStorage";

    const { Text } = Typography;

    interface SplitPaneContainerProps {
      /** 左侧面板内容 */
      leftPanel: React.ReactNode;
      /** 右侧面板内容 */
      rightPanel: React.ReactNode;
      /** 左侧面板标题 */
      leftTitle?: string;
      /** 右侧面板标题 */
      rightTitle?: string;
      /** 最小面板宽度百分比 (默认 20) */
      minLeftSize?: number;
      /** 最大面板宽度百分比 (默认 80) */
      maxLeftSize?: number;
    }

    const STORAGE_KEY = "mdTranslatorSplitRatio";

    /**
     * 分屏容器组件
     * - 默认 50:50 比例
     * - 细线分隔条带 hover 高亮
     * - 比例通过 localStorage 持久化
     */
    const SplitPaneContainer: React.FC<SplitPaneContainerProps> = ({
      leftPanel,
      rightPanel,
      leftTitle,
      rightTitle,
      minLeftSize = 20,
      maxLeftSize = 80,
    }) => {
      const t = useTranslations("common");

      // 持久化的比例，格式 "50,50"
      const [savedRatio, setSavedRatio] = useLocalStorage(STORAGE_KEY, "50,50");

      // 将持久化的比例转换为 sizes 数组
      const getInitialSizes = useCallback((): number[] => {
        if (savedRatio) {
          const parts = savedRatio.split(",");
          if (parts.length === 2) {
            const left = parseInt(parts[0], 10);
            const right = parseInt(parts[1], 10);
            if (!isNaN(left) && !isNaN(right) && left + right === 100) {
              return [left, right];
            }
          }
        }
        return [50, 50];
      }, [savedRatio]);

      const [sizes, setSizes] = useState<number[]>(getInitialSizes);

      // 比例变化时保存到 localStorage
      const handleSplitChange = useCallback((newSizes: number[]) => {
        setSizes(newSizes);
        setSavedRatio(`${newSizes[0]},${newSizes[1]}`);
      }, [setSavedRatio]);

      // 恢复初始比例 (50:50)
      const handleResetRatio = useCallback(() => {
        setSizes([50, 50]);
        setSavedRatio("50,50");
      }, [setSavedRatio]);

      return (
        <Card className="h-full">
          <Flex vertical className="h-full" gap="small">
            {/* 分屏头部 - 显示标题和重置按钮 */}
            <Flex justify="space-between" align="center">
              <Space>
                {leftTitle && (
                  <Text strong className="text-sm">
                    {leftTitle}
                  </Text>
                )}
                {rightTitle && (
                  <Text strong className="text-sm">
                    {rightTitle}
                  </Text>
                )}
              </Space>
              <Text
                type="secondary"
                className="text-xs cursor-pointer hover:text-blue-500"
                onClick={handleResetRatio}
              >
                {t("resetRatio") || "重置比例"}
              </Text>
            </Flex>

            {/* 分屏主体 */}
            <div className="flex-1 min-h-0">
              <Split
                className="flex h-full"
                sizes={sizes}
                minSize={minLeftSize}
                maxSize={maxLeftSize}
                onChange={handleSplitChange}
                gutterSize={8}
                gutterAlign="center"
                direction="horizontal"
                gutterStyle={() => ({
                  backgroundColor: "#f0f0f0",
                  cursor: "col-resize",
                  width: "8px",
                })}
                draggableDotStyle={() => ({
                  width: "4px",
                  height: "40px",
                  backgroundColor: "#d9d9d9",
                  borderRadius: "2px",
                })}
                renderDragDraggingLine={(_index, status) => ({
                  backgroundColor: status.isDragging ? "#1890ff" : "#d9d9d9",
                })}
              >
                <div className="h-full overflow-hidden">{leftPanel}</div>
                <div className="h-full overflow-hidden">{rightPanel}</div>
              </Split>
            </div>
          </Flex>
        </Card>
      );
    };

    export default SplitPaneContainer;
    ```

    注意: react-split 的具体 API 需要根据实际安装版本调整。上述代码使用 react-split@1.x 的 API 风格。

    **关键实现点:**
    - 使用 `react-split` 的 `Split` 组件实现水平分屏
    - `sizes` 数组控制左右比例，默认 [50, 50]
    - `gutterSize={8}` 设置分隔条宽度为 8px
    - `direction="horizontal"` 设置水平分屏
    - 分隔条样式使用细线设计 (8px 宽，hover 高亮)
    - 通过 `useLocalStorage` hook 持久化比例到 localStorage
