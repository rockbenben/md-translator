    ```typescript
    "use client";

    import React from "react";
    import ReactMarkdown from "react-markdown";
    import remarkGfm from "remark-gfm";
    import remarkMath from "remark-math";
    import rehypeKatex from "rehype-katex";
    import { Prism as SyntaxHighlighter } from "react-syntax-highlighter";
    // 使用深色主题
    import { oneDark } from "react-syntax-highlighter/dist/esm/styles/prism";
    import "katex/dist/katex.min.css"; // KaTeX CSS

    interface MarkdownPreviewProps {
      /** Markdown 内容 */
      content: string;
      /** 可选的 className */
      className?: string;
    }

    /**
     * Markdown 预览组件
     * - 支持 Basic Markdown (标题、列表、代码块、链接、粗体、斜体)
     * - 支持 GFM (表格、任务列表、删除线)
     * - 支持 LaTeX 数学公式 ($x$ 和 $$x$$)
     * - 代码块使用 Prism 语法高亮
     */
    const MarkdownPreview: React.FC<MarkdownPreviewProps> = ({ content, className = "" }) => {
      return (
        <div className={`markdown-preview ${className}`}>
          <ReactMarkdown
            remarkPlugins={[remarkGfm, remarkMath]}
            rehypePlugins={[rehypeKatex]}
            components={{
              // 自定义代码块渲染 - 使用语法高亮
              code({ node, className, children, ...props }) {
                const match = /language-(\w+)/.exec(className || "");
                const inline = !match && !className;

                if (inline) {
                  return (
                    <code className={`${className} px-1 py-0.5 bg-gray-100 dark:bg-gray-800 rounded text-sm`} {...props}>
                      {children}
                    </code>
                  );
                }

                return (
                  <SyntaxHighlighter
                    style={oneDark}
                    language={match ? match[1] : "text"}
                    PreTag="div"
                    customStyle={{
                      margin: "1em 0",
                      borderRadius: "0.5em",
                      fontSize: "0.875em",
                    }}
                    {...props}
                  >
                    {String(children).replace(/\n$/, "")}
                  </SyntaxHighlighter>
                );
              },
              // 自定义链接 - 打开新窗口
              a({ node, children, href, ...props }) {
                return (
                  <a href={href} target="_blank" rel="noopener noreferrer" {...props}>
                    {children}
                  </a>
                );
              },
              // 自定义表格 - 响应式
              table({ node, children, ...props }) {
                return (
                  <div className="overflow-x-auto">
                    <table className="min-w-full border border-gray-200 dark:border-gray-700" {...props}>
                      {children}
                    </table>
                  </div>
                );
              },
            }}
          >
            {content}
          </ReactMarkdown>
        </div>
      );
    };

    export default MarkdownPreview;
    ```

    **关键实现点:**
    - 使用 `react-markdown` 作为核心渲染库
    - 使用 `remark-gfm` 支持 GitHub Flavored Markdown

    - 使用 `react-syntax-highlighter` 的 Prism 实现代码语法高亮
    - 导入 `katex/dist/katex.min.css` 以正确渲染 LaTeX

    - 自定义 `a` 组件添加 `target="_blank"`
