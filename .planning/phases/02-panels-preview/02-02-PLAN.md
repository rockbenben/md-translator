        </div>
      </Card>

          <TextArea

            className="flex-1"
            aria-label={t("sourceArea")}
          />
          <Flex justify="end">
            <Text type="secondary" className="!text-xs">

          </Flex>
        </Flex>
      </Card>
    );

    // 右侧面板内容 - 根据预览内容和预览模式选择

        title={
          <Flex justify="space-between" align="center">

            <Button

              icon={previewContent === 'source' ? <FileTextOutlined /> : <PreviewOutlined />}

            </Button>
          </Flex>
        }
      >

            </Text>
          </Flex>
        </Flex>
      </Card>
