1. **导入新增:**
```typescript
import { SyncOutlined, SyncOutlined } from '@ant-design/icons';
import { useScrollSync } from './useScrollSync';
import { useLocalStorage } from '@/app/hooks/useLocalStorage';
```

2. **添加状态和 refs:**
```typescript
// localStorage key for sync preference
const SYNC_STORAGE_KEY = 'mdTranslatorScrollSync';

// Scroll sync state (default: true per D-02)
const [scrollSyncEnabled, setScrollSyncEnabled] = useLocalStorage(SYNC_STORAGE_KEY, true);

// Refs for scroll containers - these will be attached to scrollable elements
const leftScrollRef = useRef<HTMLDivElement>(null);
const rightScrollRef = useRef<HTMLDivElement>(null);

// Initialize useScrollSync hook
const { sourceRef: sourceScrollRef, targetRef: targetScrollRef, setEnabled: setSyncEnabled } = useScrollSync({
  enabled: scrollSyncEnabled,
  syncDirection: 'bidirectional',
  useRAF: true,
});
```

3. **修改 leftPanel 和 rightPanel:**
将 Card 的 className="h-full" 包裹在 inner div with ref:
```tsx
const leftPanel = (
  <Card title={t("sourceArea")}>
    <div ref={leftScrollRef} className="flex-1 overflow-auto h-[400px]">
      {/* content */}
    </div>
  </Card>
);
```

4. **添加工具栏按钮:**
在现有按钮旁边添加切换按钮:
```tsx
<Button
  icon={scrollSyncEnabled ? <SyncOutlined /> : <SyncOutlined />}
  onClick={() => setScrollSyncEnabled(!scrollSyncEnabled)}
  type={scrollSyncEnabled ? 'primary' : 'default'}
  title={scrollSyncEnabled ? t("syncScrollOn") : t("syncScrollOff")}
>
  {scrollSyncEnabled ? t("syncOn") : t("syncOff")}
</Button>
```

5. **建立面板引用映射:**
在 useEffect 中建立关联:
```typescript
useEffect(() => {
  // Map scroll refs to useScrollSync refs
  if (leftScrollRef.current) {
    (sourceScrollRef as React.MutableRefObject<HTMLElement | null>).current = leftScrollRef.current;
  }
  if (rightScrollRef.current) {
    (targetScrollRef as React.MutableRefObject<HTMLElement | null>).current = rightScrollRef.current;
  }
}, [sourceScrollRef, targetScrollRef]);
```

注意: 实际实现时，useScrollSync 的 sourceRef/targetRef 应该直接赋值给容器的 ref 属性，而不是通过 useEffect 桥接。

调整 useScrollSync hook 以接受外部容器 refs，使 SplitPaneView 可以直接赋值:

修改 useScrollSync 接受容器 ref 作为参数:
```typescript
interface ScrollSyncOptions {
  /** Ref to source scroll container */
  sourceRef: RefObject<HTMLElement | null>;
  /** Ref to target scroll container */
  targetRef: RefObject<HTMLElement | null>;
  /** Enable/disable sync (controlled externally) */
  enabled?: boolean;
  /** Sync direction */
  syncDirection?: 'bidirectional' | 'source-to-target' | 'target-to-source';
  /** RAF batching */
  useRAF?: boolean;
}
```

然后在 SplitPaneView 中直接赋值:
```tsx
<div ref={sourceScrollRef} className="flex-1 overflow-auto h-[400px]">
<div ref={targetScrollRef} className="flex-1 overflow-auto h-[400px]">
```

这样更干净，避免 useEffect 桥接。
